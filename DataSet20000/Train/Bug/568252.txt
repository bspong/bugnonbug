I have a html page with 2 select (multi-select) boxes next to each other and 2 button between them. One button has a < and the other has > for the values. 

when the page loads one select is filled with options that I had mistakenly forgot to include the attribute "value=", but I still had the value. 

For example:
<option 1234 title='a title'>Customer Name</option>
-- note the 1234, which is the record number in a db table. 

When I click a button to move the selected item to the other select box the above code fragment works.

The button makes use of a onclick event to run a javascript function to access the selected source options and move them to the destination select. In the process of doing this it properly reads the 1234 as a value when I create each new option. The javascript makes no error or warning when the value is filled in. 

While this is fine behaviour, it is still misleading. IF this was a POST my mistake would have been obvious as the select would have sent the options text only.

Reproducible: Always

Actual Results:  
bad html works

Expected Results:  
a warning should have been thrown, maybe an error.

here is a fragment of the code I am using.

=================== js
function add_items()
{
	var src_list = document.getElementById('rma_list');
	var ship_list = document.getElementById('ship_list');
	var vendor = document.getElementById('vendor').value;
	var cnt = src_list.length;
	var purge = new Array();
	for(var i=0;i<cnt;i++)
	{
		if(src_list.options[i].selected)
		{
			if(src_list.options[i].value!=0)
			{
				var title_array = src_list.options[i].title.split(' ');
				var title_vendor = title_array[0].split(':');
				if(vendor.length==0)
				{
					document.getElementById('vendor').value=title_vendor[1];
					vendor = document.getElementById('vendor').value;
					ship_list.options[ship_list.length]=new Option(src_list.options[i].text,src_list.options[i].value);
					ship_list.options[ship_list.length-1].title=src_list.options[i].title;
					purge[purge.length]=src_list.options[i].text;
				}
				else
				{
					if(vendor==title_vendor[1])
					{
						ship_list.options[ship_list.length]=new Option(src_list.options[i].text,src_list.options[i].value);
						ship_list.options[ship_list.length-1].title=src_list.options[i].title;
						purge[purge.length]=src_list.options[i].text;
					}
				}
			}
		}
	}
	for(var i=0;i<purge.length;i++)
	{
		for(var n=0;n<src_list.length;n++)
		{
			if(src_list.options[n].text==purge[i])
			{
				src_list.remove(n);
				break;
			}
		}
	}
}
